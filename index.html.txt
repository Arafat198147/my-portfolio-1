        async function adminLogin() {
            const pass = document.getElementById("admin-password-input").value;
            if (!pass) return;
            if (!GOOGLE_SCRIPT_URL || GOOGLE_SCRIPT_URL === "YOUR_GOOGLE_SCRIPT_WEB_APP_URL") {
                alert("প্রথমে GOOGLE_SCRIPT_URL সেটআপ করুন (কোডের উপরের দিকে)।");
                return;
            }
            try {
                const res = await fetch(GOOGLE_SCRIPT_URL, {
                    method: "POST",
                    body: JSON.stringify({ action: "login", password: pass })
                });
                const json = await res.json();
                if (json.success) {
                    ADMIN_PASSWORD_CACHE = pass;
                    sessionStorage.setItem("rs_admin_pass", pass);
                    refreshAdminUI();
                    closeAdminLogin();
                } else {
                    document.getElementById("admin-login-error").classList.remove("hidden");
                }
            } catch (err) {
                alert("লগইন করতে সমস্যা হয়েছে। ইন্টারনেট বা স্ক্রিপ্ট URL চেক করুন।");
            }
        }

        function adminLogout() {
            ADMIN_PASSWORD_CACHE = null;
            sessionStorage.removeItem("rs_admin_pass");
            refreshAdminUI();
        }

        let currentFormSheet = null;
        let currentFormItem = null;

        function openItemForm(sheetName, item) {
            if (!isAdminLoggedIn()) { alert("প্রথমে অ্যাডমিন লগইন করুন।"); return; }
            currentFormSheet = sheetName;
            currentFormItem = item || null;
            document.getElementById("item-form-title").innerText = item ? "এডিট করুন" : "নতুন আইটেম যোগ করুন";

            const fields = SHEET_FIELD_CONFIG[sheetName];
            const container = document.getElementById("item-form-fields");
            container.innerHTML = "";
            fields.forEach(f => {
                const value = item ? (item[f.key] || "") : "";
                let inputHtml = "";
                if (f.type === "textarea" || f.type === "textarea-lg") {
                    const rows = f.type === "textarea-lg" ? 8 : 3;
                    inputHtml = `<textarea class="form-input" rows="${rows}" data-field="${f.key}">${escapeHtml(value)}</textarea>`;
                } else if (f.type === "list") {
                    const listVal = Array.isArray(value) ? value.join("\\n") : "";
                    inputHtml = `<textarea class="form-input" rows="4" data-field="${f.key}" data-type="list">${escapeHtml(listVal)}</textarea>`;
                } else {
                    inputHtml = `<input type="text" class="form-input" data-field="${f.key}" value="${escapeHtml(String(value))}">`;
                }
                container.innerHTML += `<div><label class="form-label">${f.label}</label>${inputHtml}</div>`;
            });

            showModal("item-form-modal");
        }

        function escapeHtml(str) {
            return String(str).replace(/&/g, "&amp;").replace(/"/g, "&quot;").replace(/</g, "&lt;").replace(/>/g, "&gt;");
        }

        function closeItemForm() { hideModal("item-form-modal"); }

        async function submitItemForm() {
            const container = document.getElementById("item-form-fields");
            const inputs = container.querySelectorAll("[data-field]");
            const item = currentFormItem ? { ...currentFormItem } : {};
            inputs.forEach(inp => {
                const key = inp.dataset.field;
                if (inp.dataset.type === "list") {
                    item[key] = inp.value.split("\\n").map(s => s.trim()).filter(Boolean);
                } else {
                    item[key] = inp.value;
                }
            });

            const saveBtn = document.getElementById("item-form-save-btn");
            saveBtn.disabled = true;
            saveBtn.innerText = "সেভ হচ্ছে...";

            try {
                const action = currentFormItem ? "updateItem" : "addItem";
                const result = await callAdminAPI({ action, sheet: currentFormSheet, item });
                if (result.success) {
                    closeItemForm();
                    await loadData();
                } else {
                    alert("সেভ করতে সমস্যা: " + (result.error || "অজানা সমস্যা"));
                }
            } catch (err) {
                alert("নেটওয়ার্ক সমস্যা, আবার চেষ্টা করুন।");
            }
            saveBtn.disabled = false;
            saveBtn.innerText = "সেভ করুন";
        }

        async function deleteItem(sheetName, id) {
            if (!isAdminLoggedIn()) { alert("প্রথমে অ্যাডমিন লগইন করুন।"); return; }
            if (!confirm("আপনি কি নিশ্চিত এই আইটেমটি ডিলিট করতে চান?")) return;
            try {
                const result = await callAdminAPI({ action: "deleteItem", sheet: sheetName, id });
                if (result.success) {
                    await loadData();
                } else {
                    alert("ডিলিট করতে সমস্যা: " + (result.error || "অজানা সমস্যা"));
                }
            } catch (err) {
                alert("নেটওয়ার্ক সমস্যা, আবার চেষ্টা করুন।");
            }
        }

        function openProfileForm() {
            if (!isAdminLoggedIn()) { alert("প্রথমে অ্যাডমিন লগইন করুন।"); return; }
            currentFormSheet = "__PROFILE__";
            currentFormItem = { ...DATA.profile };
            document.getElementById("item-form-title").innerText = "যোগাযোগের তথ্য এডিট করুন";

            const fields = [
                { key: "profileName", label: "নাম" },
                { key: "profileEmail", label: "ইমেইল" },
                { key: "profileWhatsapp", label: "হোয়াটসঅ্যাপ নম্বর (কান্ট্রি কোড সহ, যেমন +8801XXXXXXXXX)" },
                { key: "profileFacebookUrl", label: "ফেসবুক প্রোফাইল লিংক" },
                { key: "profileMessengerUrl", label: "মেসেঞ্জার লিংক" },
                { key: "profileAvatar", label: "প্রোফাইল ছবির লিংক" }
            ];
            const container = document.getElementById("item-form-fields");
            container.innerHTML = "";
            fields.forEach(f => {
                const value = DATA.profile[f.key] || "";
                container.innerHTML += `<div><label class="form-label">${f.label}</label><input type="text" class="form-input" data-field="${f.key}" value="${escapeHtml(value)}"></div>`;
            });
            showModal("item-form-modal");
        }

        // প্রোফাইল সেভ (submitItemForm ফাংশন এক্সটেন্ড করে)
        const originalSubmit = submitItemForm;
        submitItemForm = async function() {
            if (currentFormSheet === "__PROFILE__") {
                const container = document.getElementById("item-form-fields");
                const inputs = container.querySelectorAll("[data-field]");
                const profile = {};
                inputs.forEach(inp => profile[inp.dataset.field] = inp.value);
                const saveBtn = document.getElementById("item-form-save-btn");
                saveBtn.disabled = true; saveBtn.innerText = "সেভ হচ্ছে...";
                try {
                    const result = await callAdminAPI({ action: "updateProfile", profile });
                    if (result.success) { closeItemForm(); await loadData(); }
                    else alert("সেভ করতে সমস্যা: " + (result.error || ""));
                } catch (err) { alert("নেটওয়ার্ক সমস্যা।"); }
                saveBtn.disabled = false; saveBtn.innerText = "সেভ করুন";
                return;
            }
            return originalSubmit();
        };

            submitBtn.disabled = true;
            submitBtn.innerHTML = `<span>মেসেজ পাঠানো হচ্ছে...</span> <i class="fa-solid fa-spinner animate-spin"></i>`;

            const email = document.getElementById("form-email").value;
            const message = document.getElementById("form-message").value;

            if (!GOOGLE_SCRIPT_URL || GOOGLE_SCRIPT_URL === "YOUR_GOOGLE_SCRIPT_WEB_APP_URL") {

            try {
                await fetch(GOOGLE_SCRIPT_URL, {
                    method: "POST",
                    body: JSON.stringify({ action: "submitLead", lead: { name, phone, email, message } })
                });
